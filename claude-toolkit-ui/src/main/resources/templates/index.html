Quick Start
# 1. Clone & Build
git clone https://github.com/YOUR_USERNAME/claude-java-toolkit.git
cd claude-java-toolkit
mvn install -DskipTests

# 2. API 키 설정
export CLAUDE_API_KEY=sk-ant-...         # Mac/Linux
set CLAUDE_API_KEY=sk-ant-...            # Windows

# 3. UI 실행
cd claude-toolkit-ui
mvn spring-boot:run

# → http://localhost:8027
